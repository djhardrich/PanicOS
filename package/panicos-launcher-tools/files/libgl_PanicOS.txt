#!/bin/bash
#
# PanicOS — use system mesa3d (panfrost + Wayland EGL) instead of bundled gl4es.
# Our mesa3d ships full OpenGL + EGL with Wayland compositor support in /usr/lib.
# We intentionally skip the gl4es LD_LIBRARY_PATH block from libgl_default.txt;
# ports that bundle gl4es 1.1.7 get system mesa3d instead, which actually works
# on the H700's split DRM topology (card0=panfrost, card1=sun4i-drm display).
#
# Port-specific non-GL libraries in ./libs.$DEVICE_ARCH or ./libs are still loaded.

if [ -d "$PWD/libs.$DEVICE_ARCH" ]; then
  export LD_LIBRARY_PATH="$PWD/libs.$DEVICE_ARCH:$LD_LIBRARY_PATH"
elif [ -d "$PWD/libs" ]; then
  export LD_LIBRARY_PATH="$PWD/libs:$LD_LIBRARY_PATH"
fi
