#!/bin/bash
#
# Modular - PanicOS
#
# Sourced by PortMaster's outer PortMaster.sh:
#     [ -f "${controlfolder}/mod_${CFW_NAME}.txt" ] && source "${controlfolder}/mod_${CFW_NAME}.txt"
# CFW_NAME is set to "PanicOS" by device_info.txt's OS_NAME-fallback
# branch (uOS/JELOS/ROCKNIX-style detection — our /etc/os-release sets
# OS_NAME="PanicOS").
#
# Mirrors mod_ROCKNIX.txt's runtime contract: sets driver + Godot opts
# and defines pm_platform_helper, which port launchers call after
# spawning their child process so sway can grab the new window and
# fullscreen it. /usr/bin/portmaster_sway_fullscreen.sh is shipped by
# the panicos-sway package and uses our sway_fullscreen bash function
# (sourced from /etc/profile.d/sway-fullscreen.sh).

PORT_32BIT="${PORT_32BIT:-N}"

# ── westonwrap runtime patcher ──────────────────────────────────────────────
# Applies two patches to westonwrap.sh (from weston_pkg squashfs) at runtime
# so we don't own the squashfs and survive westonwrap updates.
#
# Fix 1 — waydisp="wayland-5": hardcoded socket name; sway listens on wayland-1.
#
# Fix 2 — panfrost bypass: mirrors ROCKNIX's Panfrost/SD bypass. Instead of
# starting a nested weston + crusty EGL (which can't open a GBM context on
# the H700's split DRM topology), we skip weston entirely and run the app
# directly on the existing sway Wayland session. System mesa (panfrost) has
# proper Wayland EGL support; the bundled crusty_x11egl does not.
# Injected after check_gptokeyb, before weston setup, via Python string replace.
#
# _PANICOS_WESTON_PATCHED guards against infinite re-exec.
export LIBSEAT_BACKEND=builtin
if [[ -z "$_PANICOS_WESTON_PATCHED" ]] && [[ "${BASH_SOURCE[1]:-}" == *westonwrap* ]]; then
    export _PANICOS_WESTON_PATCHED=1
    _patched="/tmp/westonwrap-panicos.sh"
    cp "${BASH_SOURCE[1]}" "$_patched"
    chmod +x "$_patched"
    sed -i 's/waydisp="wayland-5"/waydisp="${WAYLAND_DISPLAY:-wayland-1}"/g' "$_patched"
    python3 - "$_patched" <<'PYEOF'
import sys, re
txt = open(sys.argv[1]).read()
bypass = r"""
# PanicOS panfrost bypass: skip weston, connect app directly to sway
# Mirrors ROCKNIX's Panfrost/SD bypass path in westonwrap.
if [ -n "$WAYLAND_DISPLAY" ] && [ -S "${XDG_RUNTIME_DIR:-/var/run/0-runtime-dir}/$WAYLAND_DISPLAY" ]; then
    echo "PanicOS: panfrost/sway detected, bypassing weston (socket: $WAYLAND_DISPLAY)"
    export waydisp="$WAYLAND_DISPLAY"
    export xsock="${DISPLAY:-:0}"
    export app_command="${@:5}"
    unset WRAPPED_LIBRARY_PATH_MALI WRAPPED_PRELOAD_MALI
    unset WRAPPED_LIBRARY_PATH_PANFROST WRAPPED_PRELOAD_PANFROST
    run_user_command
    cleanup_and_exit
fi
"""
# Match the standalone (unindented) call only — westonwrap has an indented
# occurrence inside run_user_command() that must not receive the injection.
txt = re.sub(r'^check_gptokeyb$',
             'check_gptokeyb\n' + bypass.rstrip(),
             txt, count=1, flags=re.MULTILINE)
open(sys.argv[1], 'w').write(txt)
PYEOF
    exec bash "$_patched" "$@"
fi
# ────────────────────────────────────────────────────────────────────────────

# Mesa3D driver path — aarch64 only, no /usr/lib32 (PanicOS doesn't
# ship 32-bit compat libs).
export LIBGL_DRIVERS_PATH=/usr/lib/dri

# Godot resolution flags used by Godot 2 / 3+ ports respectively.
GODOT2_OPTS="-r ${DISPLAY_WIDTH}x${DISPLAY_HEIGHT} -f"
GODOT_OPTS="--resolution ${DISPLAY_WIDTH}x${DISPLAY_HEIGHT} -f"

pm_platform_helper() {
    if [ -e "$PM_PIPE" ]; then
        PortMasterDialogExit
    fi
    if [ -e "/usr/bin/portmaster_sway_fullscreen.sh" ]; then
        /usr/bin/portmaster_sway_fullscreen.sh "$(basename "$1")"
    fi
}

# Override PortMaster's get_controls. The upstream get_controls() in
# control.txt hardcodes /dev/input/by-path/platform-* device-detection
# (Anbernic, OdroidGo, GameForce, RG552, GKD-Pixel, etc.) — none of which
# match our rocknix-joypad. With no match, DEVICE falls through to empty,
# so PortMaster writes an EMPTY /tmp/gamecontrollerdb.txt. SDL2 then
# falls back to its built-in default mapping (Xbox-positional A=south),
# which contradicts our system gamecontrollerdb (Nintendo-positional
# a:b1, b:b0 — A=east, the right physical button), and every PortMaster
# port comes up with A/B/X/Y reversed.
#
# ROCKNIX replaces get_controls with one that runs their mapper.txt to
# synthesize the db from EmulationStation config. We don't have ES, so
# we just stuff our system-wide gamecontrollerdb into /tmp directly.
# Sourced AFTER control.txt and BEFORE the wrapper calls get_controls,
# so this definition wins.
get_controls() {
    export SDL_GAMECONTROLLERCONFIG_FILE="/tmp/gamecontrollerdb.txt"
    if [ -f /usr/share/SDL-GameControllerDB/gamecontrollerdb.txt ]; then
        cp -f /usr/share/SDL-GameControllerDB/gamecontrollerdb.txt \
              "${SDL_GAMECONTROLLERCONFIG_FILE}"
    else
        : > "${SDL_GAMECONTROLLERCONFIG_FILE}"
    fi
    sdl_controllerconfig="$(< "${SDL_GAMECONTROLLERCONFIG_FILE}")"
}
